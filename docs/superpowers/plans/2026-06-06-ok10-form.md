### Task 3: Create OK10Form Component Structure

**Files:**

```tsx
/**
 * OK10 Form (Obrachunek wg kroku 10 AA) component.
 */
import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import useForm from '../../hooks/useForm';
import CollapsibleList from '../Common/CollapsibleList';
import { getFeelings, getNeeds, FeelingsData, NeedsData } from '../../services/reference';

interface OK10FormProps {
  formId?: string | null;
}

export default function OK10Form({ formId = null }: OK10FormProps) {
  const navigate = useNavigate();
  const { formData, updateField, submitForm, isSaving, lastSaved, error } = useForm(formId, 'OK10');

  // Load NVC reference data
  const [feelings, setFeelings] = useState<FeelingsData>({ fulfilled: {}, unfulfilled: {} });
  const [needs, setNeeds] = useState<NeedsData>({});
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function loadData() {
      try {
        const [feelingsData, needsData] = await Promise.all([
          getFeelings(),
          getNeeds()
        ]);
        setFeelings(feelingsData);
        setNeeds(needsData);
      } catch (err) {
        console.error('Failed to load reference data:', err);
      } finally {
        setLoading(false);
      }
    }
    loadData();
  }, []);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    await submitForm();
    navigate('/dashboard');
  };

  if (loading) {
    return <div style={{ maxWidth: '1000px', margin: '0 auto', padding: '20px' }}>Ładowanie...</div>;
  }

  // Merge fulfilled and unfulfilled feelings for display
  const allFeelings = {
    ...feelings.fulfilled,
    ...feelings.unfulfilled,
  };

  return (
    <div style={{ maxWidth: '1000px', margin: '0 auto', padding: '20px' }}>
      <h1>OK10 - Obrachunek wg kroku 10 AA</h1>

      {error && (
        <div style={{
          color: '#d32f2f',
          background: '#ffebee',
          padding: '10px',
          borderRadius: '4px',
          marginBottom: '15px'
        }}>
          {error}
        </div>
      )}

      {lastSaved && (
        <div style={{
          color: '#2e7d32',
          fontSize: '14px',
          marginBottom: '15px'
        }}>
          Ostatnio zapisano: {lastSaved.toLocaleTimeString('pl-PL')}
        </div>
      )}

      <form onSubmit={handleSubmit}>
        {/* Form fields will be added in next steps */}
        
        <div style={{ display: 'flex', gap: '10px', justifyContent: 'center', marginTop: '30px' }}>
          <button
            type="button"
            onClick={() => navigate('/dashboard')}
            style={{
              padding: '12px 24px',
              background: '#f5f5f5',
              border: '1px solid #ccc',
              borderRadius: '4px',
              cursor: 'pointer',
              fontSize: '13px',
            }}
          >
            Anuluj
          </button>
          <button
            type="submit"
            disabled={isSaving}
            style={{
              padding: '12px 24px',
              background: '#2196f3',
              color: 'white',
              border: 'none',
              borderRadius: '4px',
              cursor: isSaving ? 'not-allowed' : 'pointer',
              fontSize: '13px',
              opacity: isSaving ? 0.6 : 1,
            }}
          >
            {isSaving ? 'Zapisywanie...' : 'Zapisz'}
          </button>
        </div>
      </form>
    </div>
  );
}
```
